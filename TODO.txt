class CustomPrintPreviewDialog : public QDialog {
    Q_OBJECT
public:
    CustomPrintPreviewDialog(QPrinter* printer, QWidget* parent = nullptr)
        : QDialog(parent), m_printer(printer) {
        auto* layout = new QVBoxLayout(this);

        // Add your custom fields
        auto* optionsLayout = new QHBoxLayout();
        m_headerField = new QLineEdit();
        m_footerField = new QLineEdit();
        optionsLayout->addWidget(new QLabel("Header:"));
        optionsLayout->addWidget(m_headerField);
        optionsLayout->addWidget(new QLabel("Footer:"));
        optionsLayout->addWidget(m_footerField);

        // add margins

        layout->addLayout(optionsLayout);

        // The actual preview widget
        m_preview = new QPrintPreviewWidget(m_printer, this);
        layout->addWidget(m_preview);

        connect(m_preview, &QPrintPreviewWidget::paintRequested,
                this, &CustomPrintPreviewDialog::renderPreview);

        resize(900, 700);
    }

private slots:
    void renderPreview(QPrinter* printer) {
        // Render your QTextDocument here
        m_document->print(printer);
    }

private:
    QPrinter* m_printer;
    QPrintPreviewWidget* m_preview;
    QLineEdit* m_headerField;
    QLineEdit* m_footerField;
    QTextDocument* m_document;
};

Scripting ideas:

📄 Export & Formatting
- Scene Outline Export: generate a list of scene titles with summaries/tags.
- POV/Tag Filter Export: export only scenes tagged with a specified keyword.
- Timeline Export: collect scene dates/times from tags and output a chronological timeline.

🔍 Analysis & Consistency
- Word Count per Scene/Chapter: tally words and report totals for pacing analysis.
- Tag Frequency Report: count how often specific tags (e.g. POV, location) appear.
- Dialogue Ratio Check: calculate percentage of dialogue vs. narration in each scene.
- Character Mention Tracker: scan text for character names and produce a per‑scene presence map.

✍️ Text Manipulation
- Smart Find/Replace: replace words/phrases across all scenes with undo‑safe logging.
- Tag Injection: automatically add tags based on regex matches (e.g. detect “Letter:” and tag scene as correspondence).

🛠️ UI & Workflow Automation
- Menu Macro Runner: bind a script to a menu item that runs a sequence of editor commands.
- Batch Scene Reorder: reorder scenes based on tag criteria (e.g. timeline) instead of manual drag.
- Quick Outline Generator: insert a new document section with chapter/scene headings only.
- Delete Empty Scenes: scan tree and remove nodes with no text content.

🎁 Showcase / Fun
- Random Prompt Inserter: insert a writing prompt at the cursor from a supplied list.
- Mood Tagger: scan for sentiment words and tag scenes with “tone: dark/light/romantic.”
- Chapter Title Styler: auto‑apply formatting (caps, underline) to chapter headings.
